# Gradient descent
def gradient_descent(X, y, learning_rate, num_iterations):
    m, n = X.shape
    # Initialize weights and bias
    theta = np.zeros(n)
    bias = 0
    
    # Gradient descent loop
    losses = []
    for i in range(num_iterations):
        # Compute predictions
        z = np.dot(X, theta) + bias
        y_pred = sigmoid(z)
        
        # Compute gradients
        dz = y_pred - y
        dw = np.dot(X.T, dz) / m
        db = np.sum(dz) / m
        
        # Update parameters
        theta -= learning_rate * dw
        bias -= learning_rate * db
        
        # Compute loss
        loss = compute_loss(y, y_pred)
        losses.append(loss)
        
        if i % 100 == 0:
            print(f"Iteration {i}, Loss: {loss}")
    
    return theta, bias, losses

def plot_decision_boundary(X, y, model):
    # Define the grid boundaries for the plot
    x_min, x_max = X[:, 0].min() - 1, X[:, 0].max() + 1
    y_min, y_max = X[:, 1].min() - 1, X[:, 1].max() + 1
    xx, yy = np.meshgrid(np.arange(x_min, x_max, 0.1),
                         np.arange(y_min, y_max, 0.1))

    # Prepare grid to be used in model prediction
    grid = np.c_[xx.ravel(), yy.ravel()]
    grid_features = np.c_[grid[:, 0] * grid[:, 1],  # Feature_Product
                          grid[:, 0] + grid[:, 1],  # Feature_Add
                          np.divide(grid[:, 0], grid[:, 1], out=np.zeros_like(grid[:, 0]), where=grid[:, 1]!=0)]  # Feature_Divide, handling division by zero
    
    # Combine original grid features with the additional ones for prediction
    grid_combined = np.hstack((grid, grid_features))

    # Predict using the logistic regression model
    Z = model.predict(grid_combined)
    Z = Z.reshape(xx.shape)

    # Plot the decision boundary
    plt.contourf(xx, yy, Z, alpha=0.5)
    plt.colorbar()
    plt.scatter(X[:, 0], X[:, 1], c=y, edgecolors='k')
    
    plt.xlabel('Feature1')
    plt.ylabel('Feature2')
    plt.title('Decision Boundary with Logistic Regression')
    plt.show()
